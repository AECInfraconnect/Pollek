```rust
// หลังสร้าง health channel + ก่อน/หลัง spawn renewal task:
let (jwks_tx, jwks_rx) = tokio::sync::watch::channel(Vec::<serde_json::Value>::new());
let (roots_changed_tx, mut roots_changed_rx) = tokio::sync::watch::channel(0u64);

// ใช้ client ที่ auth ด้วย SVID ปัจจุบัน (mTLS) — เฉพาะ cloud mode
if self.cloud_url.starts_with("https://") {
    if let Ok(tb_client) = self.bootstrap.mtls.build_client(None) {
        let _poller = dek_spire_node::spawn_trust_bundle_poller(
            tb_client,
            self.cloud_url.clone(),
            self.bootstrap.mtls.root_ca_path.clone(),
            jwks_tx,
            roots_changed_tx,
            self.cancel.clone(),
        );
    }
}
```
> Local mode (http 127.0.0.1) ข้าม poller — local ใช้ pinned key ไม่ต้อง rotate root (invariant I2)

### Hot-rebuild mTLS เมื่อ root เปลี่ยน
DEK clients ที่ใช้ root CA (syncer/bundle-sync/telemetry/renewal) ต้อง rebuild เมื่อ `roots_changed_rx` เปลี่ยน:
```rust
// ใน renewal loop หรือ task กลาง: watch roots_changed -> rebuild + hot-swap (ArcSwap client)
tokio::spawn(async move {
    while roots_changed_rx.changed().await.is_ok() {
        let seq = *roots_changed_rx.borrow();
        info!("trust root changed (seq={seq}); rebuilding mTLS clients");
        if let Ok(new_client) = mtls.build_client(None) {
            shared_client.store(Arc::new(new_client));  // ArcSwap; ไม่ interrupt
        }
    }
});
```
> `build_client` อ่าน root จาก `root_ca_path` ที่ poller เขียนใหม่ → client ใหม่ trust root ใหม่
> JWT-SVID verifier (ผู้รับ) ใช้ `jwks_rx` ล่าสุด verify token
